int is_palindrome( char* the_string, int start_char, int end_char){

  if( start_char >= end_char )
    
    return 1;
  
  else if( the_string[ start_char ] != the_string[ end_char ] )
  
    return 0;
  
  return is_palindrome( the_string, start_char +1, end_char -1 );

}
